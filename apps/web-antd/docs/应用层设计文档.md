## 2. 路由配置方式

### 2.1 路由实例创建

```typescript
// router/index.ts
const router = createRouter({
  history:
    import.meta.env.VITE_ROUTER_HISTORY === 'hash'
      ? createWebHashHistory(import.meta.env.VITE_BASE)
      : createWebHistory(import.meta.env.VITE_BASE),
  routes,
  scrollBehavior: (to, _from, savedPosition) => {
    if (savedPosition) return savedPosition;
    return to.hash ? { behavior: 'smooth', el: to.hash } : { left: 0, top: 0 };
  },
});

// 创建路由守卫
createRouterGuard(router);
```

### 2.2 路由分类

| 类型 | 说明 | 文件位置 |
|------|------|----------|
| **coreRoutes** | 核心路由，不需要权限验证 | `routes/core.ts` |
| **dynamicRoutes** | 动态路由，需要权限验证 | `routes/modules/**/*.ts` |
| **externalRoutes** | 外部路由，不需要 Layout | 可选配置 |
| **staticRoutes** | 静态路由，需要权限验证 | 可选配置 |

### 2.6 路由守卫

```typescript
// router/guard.ts

// 通用守卫 - 页面加载进度
function setupCommonGuard(router: Router) {
  router.beforeEach((to) => {
    if (!to.meta.loaded && preferences.transition.progress) {
      startProgress();
    }
    return true;
  });
  router.afterEach((to) => {
    loadedPaths.add(to.path);
    if (preferences.transition.progress) stopProgress();
  });
}

// 权限守卫 - 登录验证、动态路由生成
function setupAccessGuard(router: Router) {
  router.beforeEach(async (to, from) => {
    const accessStore = useAccessStore();
    const userStore = useUserStore();
    const authStore = useAuthStore();

    // 1. 核心路由跳过验证
    if (coreRouteNames.includes(to.name as string)) return true;

    // 2. Token 检查
    if (!accessStore.accessToken) {
      if (to.meta.ignoreAccess) return true;
      return { path: LOGIN_PATH, query: { redirect: encodeURIComponent(to.fullPath) } };
    }

    // 3. 已生成路由检查
    if (accessStore.isAccessChecked) return true;

    // 4. 加载字典数据
    dictStore.setDictCacheByApi(getSimpleDictDataList);

    // 5. 获取用户信息
    if (!userStore.userInfo) {
      const authPermissionInfo = await authStore.fetchUserInfo();
      userInfo = authPermissionInfo.user;
    }

    // 6. 生成菜单和路由
    const { accessibleMenus, accessibleRoutes } = await generateAccess({
      roles: userRoles,
      router,
      routes: accessRoutes,
    });

    // 7. 保存路由信息
    accessStore.setAccessMenus(accessibleMenus);
    accessStore.setAccessRoutes(accessibleRoutes);
    accessStore.setIsAccessChecked(true);
  });
}
```

```typescript
// apps/web-antd/src/store/auth.ts
export const useAuthStore = defineStore('auth', () => {
  const loginLoading = ref(false);

  // 登录方法（支持多种登录方式）
  async function authLogin(
    type: 'mobile' | 'register' | 'social' | 'username',
    params: Recordable<any>,
    onSuccess?: () => Promise<void> | void,
  ) {
    // 1. 调用登录 API
    let loginResult: AuthApi.LoginResult;
    switch (type) {
      case 'mobile': loginResult = await smsLogin(params); break;
      case 'register': loginResult = await register(params); break;
      case 'social': loginResult = await socialLogin(params); break;
      default: loginResult = await loginApi(params);
    }

    // 2. 保存 Token
    accessStore.setAccessToken(loginResult.accessToken);
    accessStore.setRefreshToken(loginResult.refreshToken);

    // 3. 获取用户信息
    const fetchUserInfoResult = await fetchUserInfo();

    // 4. 跳转首页
    await router.push(userInfo.homePath || preferences.app.defaultHomePath);
  }

  // 登出方法
  async function logout(redirect: boolean = true) {
    await logoutApi(accessToken);
    resetAllStores();
    await router.replace({ path: LOGIN_PATH });
  }

  // 获取用户权限信息
  async function fetchUserInfo() {
    const authPermissionInfo = await getAuthPermissionInfoApi();
    userStore.setUserInfo(authPermissionInfo.user);
    userStore.setUserRoles(authPermissionInfo.roles);
    accessStore.setAccessMenus(authPermissionInfo.menus);
    accessStore.setAccessCodes(authPermissionInfo.permissions);
    return authPermissionInfo;
  }

  return { authLogin, logout, fetchUserInfo, loginLoading };
});
```

### 4.2 请求拦截器

```typescript
// api/request.ts

// 请求头处理
client.addRequestInterceptor({
  fulfilled: async (config) => {
    const accessStore = useAccessStore();

    // 1. 添加 Authorization
    config.headers.Authorization = formatToken(accessStore.accessToken);

    // 2. 添加语言
    config.headers['Accept-Language'] = preferences.app.locale;

    // 3. 添加租户ID
    config.headers['tenant-id'] = tenantEnable ? accessStore.tenantId : undefined;
    config.headers['visit-tenant-id'] = tenantEnable ? accessStore.visitTenantId : undefined;

    // 4. API 加密处理
    if (config.headers.isEncrypt) {
      config.data = apiEncrypt.encryptRequest(config.data);
      config.headers[apiEncrypt.getEncryptHeader()] = 'true';
    }

    return config;
  },
});
```

### 4.3 响应拦截器

```typescript
// api/request.ts

// 1. API 解密
client.addResponseInterceptor({
  fulfilled: (response) => {
    if (isEncryptResponse && typeof response.data === 'string') {
      response.data = apiEncrypt.decryptResponse(response.data);
    }
    return response;
  },
});

// 2. 响应数据格式处理
client.addResponseInterceptor(
  defaultResponseInterceptor({
    codeField: 'code',
    dataField: 'data',
    successCode: 0,
  }),

// 3. Token 过期处理
client.addResponseInterceptor(
  authenticateResponseInterceptor({
    client,
    doReAuthenticate,      // 重新认证
    doRefreshToken,        // 刷新 Token

  }),

// 4. 错误消息处理
client.addResponseInterceptor(
  errorMessageResponseInterceptor((msg, error) => {
    const errorMessage = error?.response?.data?.msg ?? '';
    if (error?.data?.code !== 401) {
      message.error(errorMessage || msg);
    }
  }),
);
```

### 4.4 API 接口定义规范

```typescript
// api/core/auth.ts

// 1. 定义类型命名空间
export namespace AuthApi {
  export interface LoginParams {
    username?: string;
    password?: string;
    captchaVerification?: string;
  }

  export interface LoginResult {
    accessToken: string;
    refreshToken: string;
    userId: number;
    expiresTime: number;
  }
}

// 2. 导出 API 函数
export async function loginApi(data: AuthApi.LoginParams) {
  return requestClient.post<AuthApi.LoginResult>('/system/auth/login', data, {
    headers: { isEncrypt: false },
  });
}

export async function logoutApi(accessToken: string) {
  return baseRequestClient.post('/system/auth/logout', {}, {
    headers: { Authorization: `Bearer ${accessToken}` },
  });
}

export async function getAuthPermissionInfoApi() {
  return requestClient.get<AuthPermissionInfo>('/system/auth/get-permission-info');
}
```

## 8. 关键设计决策

### 8.1 权限控制流程

```
用户登录
    ↓
获取 Token → 存储 accessToken/refreshToken
    ↓
获取权限信息 → 存储 menus/permissions/roles
    ↓
生成动态路由 → registerRoutes()
    ↓
路由守卫 → 权限验证
    ↓
页面访问 → 按钮权限判断
```

### 8.2 Token 刷新机制

```typescript
// 自动刷新流程
1. 请求拦截器检查 accessToken 是否即将过期
2. 如果即将过期，调用 refreshTokenApi 刷新
3. 更新 accessToken
4. 重试原请求
```

### 8.3 多租户支持

```typescript
// 租户切换流程
1. 从域名获取租户信息
2. 如果没有，使用 store 中的租户
3. 如果没有，使用列表第一个租户
4. 切换租户时刷新页面
```
